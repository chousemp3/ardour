252 template <class T>
253 struct EventAccumulator {
254 typedef T result_type;
255 template <class U>
256 result_type operator () (U first, U last) {
257 while (first != last) {
258 if (*first) {
259 return true;
260 }
261 ++first;
262 }
263 return false;
264 }
265 };
