65 template<typename Key>
66 void
67 increase_n(const Key& key, typename std::map<Key, size_t>& counts, size_t delta) {
68 typename std::map<Key, size_t>::iterator i = counts.find(key);
69 if (i != counts.end()) {
70 i->second += delta;
71 } else {
72 counts.insert(std::make_pair(key, delta));
73 }
74 }
