103class /*LIBPBD_API*/ OptionalLastValue
104{
105public:
106 typedef std::optional<R> result_type;
107
108 template <typename Iter>
109 result_type operator() (Iter first, Iter last) const {
110 result_type r;
111 while (first != last) {
112 r = *first;
113 ++first;
114 }

190class LIBPBD_API Connection : public std::enable_shared_from_this<Connection>
191{
192public:
193 Connection (SignalBase* b, PBD::EventLoop::InvalidationRecord* ir)
194 : _signal (b)
195 , _invalidation_record (ir)
196 {
197 if (_invalidation_record) {
198 _invalidation_record->ref ();
199 }
200 }
201
202 void disconnect ()
203 {
204 Glib::Threads::Mutex::Lock lm (_mutex);
205 SignalBase* signal = _signal.exchange (0, std::memory_order_acq_rel);
206 if (signal) {
207 /* It is safe to assume that signal has not been destructed.
208 * If ~Signal d'tor runs, it will call our signal_going_away()
209 * which will block until we're done here.
210 *
211 * This will lock Signal::_mutex, and call disconnected ()
212 * or return immediately if Signal is being destructed.
213 */
214 signal->disconnect (shared_from_this ());
215 }
216 }
217
218 void disconnected ()
219 {
220 if (_invalidation_record) {
221 _invalidation_record->unref ();
222 }
223 }
224
225 void signal_going_away ()
226 {
227 /* called with Signal::_mutex held */
228 if (!_signal.exchange (0, std::memory_order_acq_rel)) {
229 /* disconnect () grabbed the signal, but signal->disconnect()
230 * has not [yet] removed the entry from the list.
231 *
232 * Allow disconnect () to complete, which will
233 * be an effective NO-OP since SignalBase::_in_dtor is true,
234 * then we can proceed.
235 */
236 Glib::Threads::Mutex::Lock lm (_mutex);
237 }
238 if (_invalidation_record) {
239 _invalidation_record->unref ();
240 }
241 }
242
243private:
244 Glib::Threads::Mutex _mutex;
245 std::atomic<SignalBase*> _signal;
246 PBD::EventLoop::InvalidationRecord* _invalidation_record;
247};

326{

432template <typename Combiner, typename R, typename... A>
433typename std::conditional_t<std::is_void_v<R>, R, typename Combiner::result_type>
434SignalWithCombiner<Combiner, R(A...)>::operator() (A... a)
435{
436#ifdef DEBUG_PBD_SIGNAL_EMISSION
437 if (_debug_emission) {
438 std::cerr << "------ Signal @ " << this << " emission process begins with " << _slots.size() << std::endl;
439 PBD::stacktrace (std::cerr, 19);
440 }
441#endif
442
443#ifdef _MSC_VER
444 /* Regarding the note (below) it was initially
445 * thought that the problem got fixed in VS2015
446 * but in fact it still persists even in VS2022 */
447
448 /* Use the older (heap based) mapping when building with MSVC.
449 * Our StackAllocator class depends on 'boost::aligned_storage'
450 * which is known to be troublesome with Visual C++ :-
451 * https://www.boost.org/doc/libs/1_65_0/libs/type_traits/doc/html/boost_typetraits/reference/aligned_storage.html
452 */
453 std::vector<Connection*> s;
454#else
455 const std::size_t nslots = 512;
456 std::vector<Connection*,PBD::StackAllocator<Connection*,nslots> > s;
457#endif
458
459 /* First, make a copy of the current connection state for us to iterate
460 * over later (the connection state may be changed by a signal handler.
461 */
462
463 {
464 Glib::Threads::Mutex::Lock lm (_mutex);
465 /* copy only the raw pointer, no need for a shared_ptr in this
466 * context, we only use the address as a lookup into the _slots
467 * container. Note: because of the use of a stack allocator,
468 * this is *unlikely* to cause any (heap) memory
469 * allocation. That will only happen if the number of
470 * connections to this signal exceeds the value of nslots
471 * defined above. As of April 2025, the maximum number of
472 * connections appears to be ntracks+1.
473 */
474 for (auto const & [connection,functor] : _slots) {
475 s.push_back (connection.get());
476 }
477 }
478
479 if constexpr (std::is_void_v<R>) {
480 slot_function_type functor;
481
482 for (auto const & c : s) {
483
484 /* We may have just called a slot, and this may have
485 * resulted in disconnection of other slots from us.
486 * The list copy means that this won't cause any
487 * problems with invalidated iterators, but we must
488 * check to see if the slot we are about to call is
489 * still on the list.
490 */
491 bool still_there = false;
492
493 {
494 Glib::Threads::Mutex::Lock lm (_mutex);
495 typename Slots::const_iterator f = std::find_if (_slots.begin(), _slots.end(), [&](typename Slots::value_type const & elem) { return elem.first.get() == c; });
496 if (f != _slots.end()) {
497 functor = f->second;
498 still_there = true;
499 }
500 }
501
502 if (still_there) {
503#ifdef DEBUG_PBD_SIGNAL_EMISSION
504 if (_debug_emission) {
505 std::cerr << "signal @ " << this << " calling slot for connection @ " << c << " of " << _slots.size() << std::endl;
506 }
507#endif
508 functor (a...);
509 } else {
510#ifdef DEBUG_PBD_SIGNAL_EMISSION
511 if (_debug_emission) {
512 std::cerr << "signal @ " << this << " connection " << c << " of " << _slots.size() << " was no longer in the slot list\n";
513 }
514#endif
515 }
516 }
517
518#ifdef DEBUG_PBD_SIGNAL_EMISSION
519 if (_debug_emission) {
520 std::cerr << "------ Signal @ " << this << " emission process ends\n";
521 }
522#endif
523 return;
524
525 } else {
526 if (s.empty()) {
527 return typename Combiner::result_type ();
528 }
529
530 /* We would like to use a stack allocator here, but for reasons
531 * not really understood, this breaks on macOS when using
532 * the custom combiner used by libs/ardour IO's
533 * PortCountChanging signal.
534 *
535 * Using a vector here is not RT-safe but a manual code
536 * inspection reveals that there are no combiner-based signals
537 * (i.e. Signals with a return value) that are ever used in RT
538 * code.
539 *
540 * The alternative is to use alloca() but that could
541 * theoretically cause stack overflows if the number of
542 * handlers for the signal is too large (it would have to be
543 * very large, however).
544 *
545 * In short, std::vector<T> is the least-bad of two bad
546 * choices, and we've chosen this because of the lack of RT use
547 * cases for a Signal with a return value.
548 *
549 */
550
551 std::vector<R> r;
552 r.reserve (s.size());
553 slot_function_type functor;
554
555 for (auto const & c : s) {
556
557 /* We may have just called a slot, and this may have resulted in
558 * disconnection of other slots from us. The list copy means that
559 * this won't cause any problems with invalidated iterators, but we
560 * must check to see if the slot we are about to call is still on the list.
561 */
562 bool still_there = false;
563
564 {
565 Glib::Threads::Mutex::Lock lm (_mutex);
566 typename Slots::const_iterator f = std::find_if (_slots.begin(), _slots.end(), [&](typename Slots::value_type const & elem) { return elem.first.get() == c; });
567
568 if (f != _slots.end()) {
569 functor = f->second;
570 still_there = true;
571 }
572 }
573 if (still_there) {
574#ifdef DEBUG_PBD_SIGNAL_EMISSION
575 if (_debug_emission) {
576 std::cerr << "signal @ " << this << " calling non-void slot for connection @ " << c << " of " << _slots.size() << std::endl;
577 }
578#endif
579 r.push_back (functor (a...));
580 }
581
582#ifdef DEBUG_PBD_SIGNAL_EMISSION
583 if (_debug_emission) {
584 std::cerr << "------ Signal @ " << this << " emission process ends\n";
585 }
586#endif
587 }
588 /* Call our combiner to do whatever is required to the result values */
589 Combiner c;
590 return c (r.begin(), r.end());
591 }
592}

594template <typename Combiner, typename R, typename... A>
595std::shared_ptr<Connection>
596SignalWithCombiner<Combiner, R(A...)>::_connect (PBD::EventLoop::InvalidationRecord* ir,
597 slot_function_type f)
598{
599 std::shared_ptr<Connection> c (new Connection (this, ir));
600 Glib::Threads::Mutex::Lock lm (_mutex);
601 _slots[c] = f;
602
603#ifdef DEBUG_PBD_SIGNAL_CONNECTIONS
604 if (_slots.size() > max_signal_subscribers) {
605 max_signal_subscribers = _slots.size();
606 }
607 if (_debug_connection) {
608 std::cerr << "+++++++ CONNECT " << this << " via connection @ " << c << " size now " << _slots.size() << std::endl;
609 stacktrace (std::cerr, 10);
610 }
611#endif
612 return c;
613}

615template <typename Combiner, typename R, typename... A>
616void
617SignalWithCombiner<Combiner, R(A...)>::disconnect (std::shared_ptr<Connection> c)
618{
619 /* ~ScopedConnection can call this concurrently with our d'tor */
620 Glib::Threads::Mutex::Lock lm (_mutex, Glib::Threads::TRY_LOCK);
621 while (!lm.locked()) {
622 if (_in_dtor.load (std::memory_order_acquire)) {
623 /* d'tor signal_going_away() took care of everything already */
624 return;
625 }
626 /* Spin */
627 lm.try_acquire ();
628 }
629 _slots.erase (c);
630 lm.release ();
631
632 c->disconnected ();
633 #ifdef DEBUG_PBD_SIGNAL_CONNECTIONS
634 if (_debug_connection) {
635 std::cerr << "------- DISCCONNECT " << this << " size now " << _slots.size() << std::endl;
636 stacktrace (std::cerr, 10);
637 }
638 #endif
639}
