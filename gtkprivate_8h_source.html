34/* The private flags that are used in the private_flags member of GtkWidget.
35 */
36typedef enum
37{
38 PRIVATE_GTK_USER_STYLE = 1 << 0,
39 PRIVATE_GTK_RESIZE_PENDING = 1 << 2,
40 PRIVATE_GTK_HAS_POINTER = 1 << 3, /* If the pointer is above a window belonging to the widget */
41 PRIVATE_GTK_SHADOWED = 1 << 4, /* If there is a grab in effect shadowing the widget */
42 PRIVATE_GTK_HAS_SHAPE_MASK = 1 << 5,
43 PRIVATE_GTK_IN_REPARENT = 1 << 6,
44 PRIVATE_GTK_DIRECTION_SET = 1 << 7, /* If the reading direction is not DIR_NONE */
45 PRIVATE_GTK_DIRECTION_LTR = 1 << 8, /* If the reading direction is DIR_LTR */
46 PRIVATE_GTK_ANCHORED = 1 << 9, /* If widget has a GtkWindow ancestor */
47 PRIVATE_GTK_CHILD_VISIBLE = 1 << 10, /* If widget should be mapped when parent is mapped */
48 PRIVATE_GTK_REDRAW_ON_ALLOC = 1 << 11, /* If we should queue a draw on the entire widget when it is reallocated */
49 PRIVATE_GTK_ALLOC_NEEDED = 1 << 12, /* If we we should allocate even if the allocation is the same */
50 PRIVATE_GTK_REQUEST_NEEDED = 1 << 13 /* Whether we need to call gtk_widget_size_request */
51} GtkPrivateFlags;

70/* Macros for setting and clearing private widget flags.
71 * we use a preprocessor string concatenation here for a clear
72 * flags/private_flags distinction at the cost of single flag operations.
73 */
74#define GTK_PRIVATE_SET_FLAG(wid,flag) G_STMT_START{ (GTK_PRIVATE_FLAGS (wid) |= (PRIVATE_ ## flag)); }G_STMT_END
75#define GTK_PRIVATE_UNSET_FLAG(wid,flag) G_STMT_START{ (GTK_PRIVATE_FLAGS (wid) &= ~(PRIVATE_ ## flag)); }G_STMT_END
