159 /* We'd like this to use bool, but there are unexplained issues using
160 * bool with a PBD::StackAllocator. They may arise from stdlib's
161 * specialiation of std::list<bool> and/or std::vector<bool>.
162 *
163 * So we use int instead, with the same semantics.
164 */
165
166 class IntBoolCombiner {
167 public:
168
169 typedef int result_type;
170
171 template <typename Iter>
172 result_type operator() (Iter first, Iter last) const {
173 int r = 0;
174 while (first != last) {
175 if (*first > 0) {
176 r = 1;
177 }
178 ++first;
179 }
180
181 return r;
182 }
183 };
