460class DisplaySuspender {
461 public:
462 DisplaySuspender() {
463 if (PublicEditor::instance()._suspend_route_redisplay_counter.fetch_add (1) == 0) {
464 PublicEditor::instance().suspend_route_redisplay ();
465 }
466 }
467 ~DisplaySuspender () {
468 if (PBD::atomic_dec_and_test (PublicEditor::instance()._suspend_route_redisplay_counter)) {
469 PublicEditor::instance().resume_route_redisplay ();
470 }
471 }
472};
