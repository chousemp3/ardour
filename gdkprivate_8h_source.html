48/* Tests whether a pair of x,y may cause overflows when converted to Pango
49 * units (multiplied by PANGO_SCALE). We don't allow the entire range, leave
50 * some space for additions afterwards, to be safe...
51 */
52#define GDK_PANGO_UNITS_OVERFLOWS(x,y) (G_UNLIKELY ( \
53 (y) >= PANGO_PIXELS (G_MAXINT-PANGO_SCALE)/2 || \
54 (x) >= PANGO_PIXELS (G_MAXINT-PANGO_SCALE)/2 || \
55 (y) <=-PANGO_PIXELS (G_MAXINT-PANGO_SCALE)/2 || \
56 (x) <=-PANGO_PIXELS (G_MAXINT-PANGO_SCALE)/2))
