89 static void peek_a_boo (std::ostream& stream) {
90#ifdef HAVE_EXECINFO
91 typename std::list<thing_with_backtrace<T>*>::iterator x;
92 for (x = all.begin(); x != all.end(); ++x) {
93 char **strings;
94 size_t i;
95
96 strings = backtrace_symbols ((*x)->allocation_backtrace, (*x)->allocation_backtrace_size);
97
98 if (strings) {
99 stream << "--- ALLOCATED SHARED_PTR @ " << (*x) << std::endl;
100 for (i = 0; i < (*x)->allocation_backtrace_size && i < 50U; i++) {
101 stream << strings[i] << std::endl;
102 }
103 free (strings);
104 }
105 }
106#else
107 stream << "execinfo not defined for this platform" << std::endl;
108#endif
109 }
