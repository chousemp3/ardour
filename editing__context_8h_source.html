187 /* NOTE: these functions assume that the "pixel" coordinate is
188 in canvas coordinates. These coordinates already take into
189 account any scrolling offsets.
190 */
191
192 samplepos_t pixel_to_sample_from_event (double pixel) const {
193
194 /* pixel can be less than zero when motion events are
195 processed. Since the pixel value is in canvas units (since
196 it comes from an event delivered to the canvas), we've
197 already run the window->canvas transform, that means that
198 the location *really* is "off to the right" and thus really
199 is "before the start".
200 */
201
202 if (pixel >= _timeline_origin) {
203 return (pixel - _timeline_origin) * samples_per_pixel;
204 } else {
205 return 0;
206 }
207 }
208
209 /* It makes no sense to ever call these functions to convert to or from a
210 * non-timeline relative pixel position, so they all assume that is
211 * what they are being asked to do.
212 */
213
214 samplepos_t pixel_to_sample (double pixel) const {
215 return pixel * samples_per_pixel;
216 }
217
218 double sample_to_pixel (samplepos_t sample) const {
219 return round (sample / (double) samples_per_pixel);
220 }
221
222 double sample_to_pixel_unrounded (samplepos_t sample) const {
223 return (sample / (double) samples_per_pixel);
224 }
