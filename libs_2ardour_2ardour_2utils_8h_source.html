119template<typename T> std::shared_ptr<AutomationControlList> route_list_to_control_list (std::shared_ptr<RouteList const> rl, std::shared_ptr<T> (Stripable::*get_control)() const) {
120 std::shared_ptr<AutomationControlList> cl (new AutomationControlList);
121 if (!rl) { return cl; }
122 for (auto const& r : *rl) {
123 std::shared_ptr<AutomationControl> ac = (r.get()->*get_control)();
124 if (ac) {
125 cl->push_back (ac);
126 }
127 }
128 return cl;
129}
